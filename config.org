** Org-Attach Configuration

Configure org-attach for per-heading attachments.
Each heading with attachments gets its own directory.

#+begin_src emacs-lisp
(after! org-attach
  ;; Keep org-attach available for existing legacy attachment: links.
  (setq org-attach-use-inheritance t
        ;; Keep Doom-style local .attach/ storage for legacy IDs.
        org-attach-id-dir ".attach/"
        org-attach-id-to-path-function-list
        '(org-attach-id-uuid-folder-format)))

;; UUID-based attachment directory for each note (matches org-roam top-level ID).
(defun my/org-root-id (&optional file)
  "Return top-level `:ID:` from FILE (or current buffer), if present."
  (with-current-buffer (or (and file (find-file-noselect file)) (current-buffer))
    (save-excursion
      (goto-char (point-min))
      (let ((limit (save-excursion
                     (if (re-search-forward "^\\*+ " nil t)
                         (match-beginning 0)
                       (point-max)))))
        (when (re-search-forward "^[ \t]*:ID:[ \t]*\\(.+\\)$" limit t)
          (string-trim (match-string-no-properties 1)))))))

(defun my/org-file-attach-dir (&optional file)
  "Return FILE's attachment directory based on top-level note ID.
Falls back to `.attach/<file-base>/` only if no top-level ID exists."
  (let* ((f (or file (buffer-file-name)
                (user-error "Current buffer is not visiting a file")))
         (dir (file-name-directory f))
         (base (file-name-base f))
         (root-id (my/org-root-id f)))
    (if root-id
        (let ((default-directory dir))
          (org-attach-dir-from-id root-id))
      (expand-file-name (concat ".attach/" base "/") dir))))

(defun my/org-open-file-attach-dir ()
  "Open this note's file-level attachment directory.
This bypasses org-attach's ID-based lookup used by `SPC m a a f`."
  (interactive)
  (unless (and (derived-mode-p 'org-mode) buffer-file-name)
    (user-error "Must be in a file-backed org-mode buffer"))
  (let ((dir (my/org-file-attach-dir buffer-file-name)))
    (make-directory dir t)
    (dired dir)))

(defun my/org-setup-file-attach-dir-h ()
  "Set `org-download-image-dir` buffer-locally to this note's attach dir."
  (when (and (derived-mode-p 'org-mode) buffer-file-name)
    (setq-local org-download-image-dir (my/org-file-attach-dir buffer-file-name))))

(add-hook 'org-mode-hook #'my/org-setup-file-attach-dir-h)

(defun my/org-move-file-attach-dir-if-needed ()
  "Move old per-file attach dir into canonical dir when needed.
Useful for notes created during transition from per-file-base naming."
  (interactive)
  (unless (and (derived-mode-p 'org-mode) buffer-file-name)
    (user-error "Must be in a file-backed org-mode buffer"))
  (let* ((base (file-name-base buffer-file-name))
         (dir (file-name-directory buffer-file-name))
         (old (expand-file-name (concat ".attach/" base "/") dir))
         (new (my/org-file-attach-dir buffer-file-name)))
    (when (and (not (string-equal old new))
               (file-directory-p old)
               (not (file-exists-p new)))
      (make-directory (file-name-directory new) t)
      (rename-file old new)
      (message "Moved attach dir: %s -> %s" old new))))

(defun my/org-clean-empty-id-dirs-in-current-file ()
  "Remove empty heading-ID attach dirs for current file.
This keeps `.attach/` tidy after migrating to file-level attachments."
  (interactive)
  (unless (and (derived-mode-p 'org-mode) buffer-file-name)
    (user-error "Must be in a file-backed org-mode buffer"))
  (let ((removed 0)
        (base-dir (file-name-directory buffer-file-name))
        (canonical (file-truename (my/org-file-attach-dir buffer-file-name))))
    (save-excursion
      (goto-char (point-min))
      (while (re-search-forward "^[ \t]*:ID:[ \t]*\\(.+\\)$" nil t)
        (let* ((id (string-trim (match-string-no-properties 1)))
               (dir (let ((default-directory base-dir))
                      (org-attach-dir-from-id id))))
          (when (and dir
                     (file-directory-p dir)
                     (not (string-equal (file-truename dir) canonical))
                     (null (directory-files dir nil "^[^.].*")))
            (delete-directory dir)
            (setq removed (1+ removed))))))
    (message "Removed %d empty ID attachment dirs" removed)
    removed))

;; Configure org-download for file-based attachments + file: links.
(after! org-download
  (setq org-download-method 'directory
        org-download-heading-lvl nil
        ;; Use plain file: links for predictable Org 9.8 preview toggling.
        org-download-link-format "[[file:%s]]\n"
        org-download-abbreviate-filename-function #'file-relative-name
        ;; Don't add annotation lines above inserted links.
        org-download-annotate-function (lambda (_link) "")))

** Window Zoom Toggle: SPC w z

Toggle a focused single-window view without destroying the layout. Saves the window configuration, switches to a single window, and restores on second call.

#+begin_src emacs-lisp
(defvar my/window-zoom--saved-config nil
  "Saved window configuration before zooming in.")

(defvar my/window-zoom--zoomed-window nil
  "The window that was zoomed, to detect if we're still in zoom state.")

(defun my/toggle-window-zoom ()
  "Toggle focused single-window view, preserving the window layout."
  (interactive)
  (if (and my/window-zoom--saved-config
           (eq (selected-window) my/window-zoom--zoomed-window)
           (= (length (window-list)) 1))
      ;; We're zoomed in and still in the same window - restore
      (progn
        (set-window-configuration my/window-zoom--saved-config)
        (setq my/window-zoom--saved-config nil
              my/window-zoom--zoomed-window nil)
        (message "Unzoomed"))
    ;; Either not zoomed, or state is stale - zoom in
    (setq my/window-zoom--saved-config (current-window-configuration)
          my/window-zoom--zoomed-window (selected-window))
    (delete-other-windows)
    (message "Zoomed")))

(map! :leader :desc "Zoom/unzoom window" "w z" #'my/toggle-window-zoom)
#+end_src

** Org Paragraph and Code Block Spacing Helpers

Ensure paragraphs are separated by a blank line in Org source, and code blocks have a blank line before and after. This keeps paragraphs visually distinct (soft-wrapped, visual-line-mode) and ensures code blocks are easy to spot and edit.

#+begin_src emacs-lisp
;; Org editing helpers: visual wrap + enforce paragraph & src-block spacing
(defun my/org--structural-line-p ()
  "Return non-nil if current line is an Org structural line we shouldn't split.
This includes headings, property-drawer markers, list markers, #+ directives,
and source/block markers."
  (let ((ln (buffer-substring-no-properties
             (line-beginning-position) (line-end-position))))
    (string-match-p
     "^[ \t]*\\(?:#\\+\\|:\\(?:PROPERTIES\\|END\\)\\|:\\s-+\\|\\*+\\s-\\|[-+*]\\s-\\|[0-9]+[.)]\\s-\\|>\\s-\\|#\\+BEGIN_SRC\\|#\\+END_SRC\\)"
     ln)))

(defun my/org-ensure-blank-lines-between-paragraphs ()
  "Manage org spacing: collapse excess blanks and handle results blocks.
Rules:
  - Collapse excessive blanks (3+) to 1 blank line
  - NO blank line between #+end_src and #+RESULTS
  - 1 blank line after #+RESULTS block ends (after result lines)"
  (interactive)
  (save-excursion
    (goto-char (point-min))
    ;; Step 1: Collapse runs of 3+ blank lines to exactly 1
    (while (re-search-forward "\n\n\n+" nil t)
      (replace-match "\n\n"))
    ;; Step 2: Remove blank line between #+end_src and #+RESULTS
    (goto-char (point-min))
    (while (re-search-forward "^[ \t]*#\\+end_src$" nil t)
      (let ((end-src-end (line-end-position)))
        (forward-line 1)
        ;; If next line is blank and line after that is #+RESULTS, delete ONLY the blank line
        (if (and (looking-at-p "^\\s-*$")
                 (save-excursion
                   (forward-line 1)
                   (looking-at-p "^[ \t]*#\\+RESULTS")))
            ;; Delete the blank line but keep the newline after #+end_src
            (delete-region (line-beginning-position) (1+ (line-end-position))))))
    ;; Step 3: Ensure blank line after #+RESULTS block ends
    (goto-char (point-min))
    (while (re-search-forward "^[ \t]*#\\+RESULTS" nil t)
      (forward-line 1)
      ;; Skip all result content lines (: prefix, blank lines, begin_example/end_example)
      (while (and (not (eobp))
                  (let ((line (buffer-substring-no-properties
                              (line-beginning-position)
                              (line-end-position))))
                    (or (string-match-p "^:" line)
                        (string-match-p "^[ \t]*$" line)
                        (string-match-p "^[ \t]*#\\+begin_example" line)
                        (string-match-p "^[ \t]*#\\+end_example" line))))
        (forward-line 1))
      ;; Now at first non-result line. Add blank if it's content (not blank, not structural)
      (unless (or (eobp)
                  (string-match-p "^\\s-*$" (buffer-substring-no-properties
                                             (line-beginning-position)
                                             (line-end-position)))
                  (string-match-p "^[ \t]*#\\+\\|^[ \t]*\\*+" (buffer-substring-no-properties
                                                               (line-beginning-position)
                                                               (line-end-position))))
        ;; Insert blank line before this content
        (goto-char (line-beginning-position))
        (insert "\n")
        (forward-line -1)))))

(defun my/org--ensure-blank-line-before (pos)
  "Ensure exactly one blank line before POS (inserts one if none)."
  (when (> pos (point-min))
    (save-excursion
      (goto-char pos)
      (forward-line -1)
      (unless (looking-at-p "^\\s-*$")
        (end-of-line)
        (insert "\n")))))

(defun my/org--ensure-blank-line-after (pos)
  "Ensure exactly one blank line after POS (inserts one if none)."
  (save-excursion
    (goto-char pos)
    (forward-line 1)
    (unless (or (eobp) (looking-at-p "^\\s-*$"))
      (insert "\n"))))

(defun my/org-add-blank-lines-around-src-blocks ()
  "Ensure a blank line exists before #+BEGIN_SRC and after #+END_SRC in the current buffer."
  (save-excursion
    (goto-char (point-min))
    (while (re-search-forward "^[ \t]*#\\+BEGIN_SRC" nil t)
      (my/org--ensure-blank-line-before (match-beginning 0)))
    (goto-char (point-min))
    (while (re-search-forward "^[ \t]*#\\+END_SRC" nil t)
      (let ((end-pos (match-end 0)))
        (save-excursion
          (goto-char end-pos)
          (forward-line 1)
          (unless (or (eobp)
                      (looking-at-p "^[ \t]*$")
                      (looking-at-p "^[ \t]*/.+/[ \t]*$"))
            (insert "\n")))))))

(defun my/org-format-buffer-on-save ()
  "Run org formatter on save, with a safe fallback during init." 
  (when (derived-mode-p 'org-mode)
    (if (fboundp 'my/org-format-buffer)
        (my/org-format-buffer)
      (progn
        (my/org-add-blank-lines-around-src-blocks)
        (my/org-ensure-blank-lines-between-paragraphs)))))

;; Org Mode hook: visual-line-mode + buffer-local before-save enforcement
(add-hook 'org-mode-hook
          (lambda ()
            ;; Use visual-line-mode so j/k traverse visual wrapped lines
            (visual-line-mode 1)
            ;; Do not turn on auto-fill: keep soft wrapping and control paragraph breaks with blank lines
            (when (boundp 'auto-fill-function) (turn-off-auto-fill))

            ;; Use one formatter entrypoint to avoid overlapping spacing passes
            (remove-hook 'before-save-hook #'my/org-add-blank-lines-around-src-blocks t)
            (remove-hook 'before-save-hook #'my/org-ensure-blank-lines-between-paragraphs t)
            (remove-hook 'before-save-hook #'my/org-format-buffer-on-save t)
            (add-hook 'before-save-hook #'my/org-format-buffer-on-save nil t)))

;; Also ensure newly-inserted src blocks get spacing immediately
(with-eval-after-load 'ob
  (advice-add 'org-babel-insert-src-block :after
              (lambda (&rest _) (my/org-add-blank-lines-around-src-blocks))))
#+end_src

Helper functions for smart theme loading - detects doom presets vs wallpaper themes:

#+begin_src emacs-lisp
(defvar my/preset-theme-cache nil
  "Cache for preset theme detection to avoid repeated JSON parsing.")
(defun my/detect-preset-theme (&optional force-refresh)
  "Detect which preset theme is currently active from pywal cache.
Returns (TYPE . NAME) where TYPE is 'doom or 'other, and NAME is the theme name.
For doom themes: NAME is like 'gruvbox-light' (without 'doom-' prefix).
For other themes: NAME is the full theme symbol like 'kuronami or 'wombat.
Returns nil if not a preset theme (wallpaper-based).
Uses caching for performance - pass FORCE-REFRESH to bypass cache."
  (let* ((cache-file (expand-file-name "~/.cache/wal/colors.json"))
         (preset-dir (expand-file-name "~/.config/wal/colorschemes/dark/")))
    (when (file-exists-p cache-file)
      ;; Check cache validity
      (if (and my/preset-theme-cache
               (not force-refresh)
               (equal (nth 5 (file-attributes cache-file))
                      (car my/preset-theme-cache)))
          ;; Return cached result
          (cdr my/preset-theme-cache)
        
        ;; Cache miss or expired - recompute
        (let* ((json-object-type 'alist)
               (json-array-type 'list)
               (current-colors (json-read-file cache-file))
               (current-bg (alist-get 'background (alist-get 'special current-colors)))
               (current-fg (alist-get 'foreground (alist-get 'special current-colors)))
               (current-c0 (alist-get 'color0 (alist-get 'colors current-colors)))
               (result nil))
          
          ;; Check all preset files for a match
          (catch 'found
            (dolist (file (directory-files preset-dir nil "\\.json$"))
              (let* ((preset-path (expand-file-name file preset-dir))
                     (preset-colors (json-read-file preset-path))
                     (preset-bg (alist-get 'background (alist-get 'special preset-colors)))
                     (preset-fg (alist-get 'foreground (alist-get 'special preset-colors)))
                     (preset-c0 (alist-get 'color0 (alist-get 'colors preset-colors))))
                
                ;; Match if bg, fg, and color0 all match
                (when (and (string= current-bg preset-bg)
                          (string= current-fg preset-fg)
                          (string= current-c0 preset-c0))
                  ;; Check if it's a doom theme or other theme
                  (setq result
                        (if (string-match "^doom-\\(.*\\)\\.json$" file)
                            ;; Doom theme: return ('doom . "gruvbox-light")
                            (cons 'doom (match-string 1 file))
                          ;; Other theme: return ('other . 'kuronami)
                          (cons 'other 
                                (intern (replace-regexp-in-string "\\.json$" "" file)))))
                  (throw 'found result)))))
          
          ;; Update cache with file timestamp and result
          (setq my/preset-theme-cache 
                (cons (nth 5 (file-attributes cache-file)) result))
          result)))))
(defun reload-pywal-theme ()
  "Reload theme - use original doom theme if preset, else use ewal.
Properly handles org-roam-ui websocket server to prevent port conflicts."
  (interactive)
  (let* ((colors-file (expand-file-name "~/.cache/wal/colors.json"))
         (json-object-type 'alist)
         (json-array-type 'list)
         (colors (json-read-file colors-file))
         (wallpaper (alist-get 'wallpaper colors))
         ;; Check if wallpaper is actual path (not "None" string)
         (is-wallpaper-based (and wallpaper 
                                  (not (string= wallpaper "None"))
                                  (file-exists-p wallpaper)))
         ;; Save org-roam-ui state before theme change
         (org-roam-ui-was-active (and (featurep 'org-roam-ui)
                                       (bound-and-true-p org-roam-ui-mode))))
    
    ;; Step 1: Stop org-roam-ui if running to prevent websocket port conflict
    (when org-roam-ui-was-active
      (org-roam-ui-mode -1)
      (message "Stopped org-roam-ui for theme reload..."))
    
    ;; Step 2: Disable all old themes first to avoid conflicts
    (mapc #'disable-theme custom-enabled-themes)
    
    ;; Step 3: Load the appropriate theme
    (if is-wallpaper-based
        ;; Wallpaper-based theme: use ewal with our fixes
        (progn
          (ewal-load-colors)
          (load-theme 'ewal-doom-one t)
          (message "Pywal theme reloaded (wallpaper-based)"))
      
      ;; No wallpaper or "None" = preset theme: try to load original theme
      ;; Force refresh cache since pywal colors just changed
      (let* ((preset-info (my/detect-preset-theme t))
             (preset-type (car preset-info))
             (preset-name (cdr preset-info)))
        (cond
         ;; Doom preset theme
         ((eq preset-type 'doom)
          (let ((theme-symbol (intern (concat "doom-" preset-name))))
            (if (member theme-symbol (custom-available-themes))
                (progn
                  (load-theme theme-symbol t)
                  (message "Loaded original doom theme: %s" theme-symbol))
              ;; Fallback to ewal if doom theme not found
              (progn
                (ewal-load-colors)
                (load-theme 'ewal-doom-one t)
                (message "Doom theme %s not found, using ewal" theme-symbol)))))
         
         ;; Other preset theme (kuronami, wombat, etc.)
         ((eq preset-type 'other)
          (if (member preset-name (custom-available-themes))
              (progn
                (load-theme preset-name t)
                (message "Loaded original theme: %s" preset-name))
            ;; Fallback to ewal if theme not found
            (progn
              (ewal-load-colors)
              (load-theme 'ewal-doom-one t)
              (message "Theme %s not found, using ewal" preset-name))))
         
         ;; No preset detected, use ewal
         (t
          (ewal-load-colors)
          (load-theme 'ewal-doom-one t)
          (message "Pywal theme reloaded (preset fallback to ewal)")))))
    
    ;; Step 4: Restart org-roam-ui if it was running before
    (when org-roam-ui-was-active
      ;; Small delay to ensure theme is fully loaded
      (run-at-time 0.3 nil 
                   (lambda ()
                     (org-roam-ui-mode 1)
                     (message "Restarted org-roam-ui after theme reload"))))))

*** Promote Heading to Concept

This function extracts a heading from a lecture or inbox note and promotes it
to a standalone concept node. The heading's contents are copied into the new
concept file under the =* Overview= section, and a reference link to the original
heading is added to the concept's =* References= section. The original heading
is preserved (not deleted).

#+begin_src emacs-lisp
  (defun my/org-roam-promote-heading-to-concept ()
    "Promote current heading into a new concept node.
This:
- Creates `concepts/<slug>.org` (if missing) with a scaffold that uses '* Overview'.
- Copies the current heading's subtree (content below the heading line) into
  the new file under '* Overview' (appends).
- Transfers the original heading's ID to the new file (if it exists), so existing
  links to that heading automatically resolve to the new concept file.
- Removes the ID from the original heading to avoid duplicates.
- Inserts a link to the new concept in the original file's '* Relations' section as a bullet point.
It does NOT delete or replace the original heading." 
    (interactive)
    (unless (org-at-heading-p)
      (user-error "Not at a heading"))
    (let* ((title (nth 4 (org-heading-components)))
           (node (org-roam-node-create :title title))
           (slug (org-roam-node-slug node))
           (file (expand-file-name (format "concepts/%s.org" slug) org-roam-directory))
           ;; original subtree bounds
           (orig-beg (save-excursion (org-back-to-heading t) (point)))
           (orig-end (save-excursion (org-end-of-subtree t t)))
           ;; body = text after the heading line up to end of subtree
           (body (save-restriction
                   (widen)
                   (when (> orig-end orig-beg)
                     (save-excursion
                       (goto-char orig-beg)
                       (forward-line 1)
                       ;; Skip properties drawer if present
                       (when (looking-at "^[ \t]*:PROPERTIES:")
                         (when (re-search-forward "^[ \t]*:END:" orig-end t)
                           (forward-line 1)))
                       ;; Skip any blank lines after properties
                       (while (and (< (point) orig-end) (looking-at-p "^\\s-*$"))
                         (forward-line 1))
                       ;; Extract the actual content
                       (let ((content-start (point)))
                         (when (< content-start orig-end)
                           (string-trim (buffer-substring-no-properties content-start orig-end)))))))))
      ;; Ensure original heading has an ID so we can link to it later if needed
      (let ((orig-id (save-excursion
                       (org-back-to-heading t)
                       (or (org-entry-get nil "ID")
                           (org-id-get-create)))))
        ;; Create scaffold if file doesn't exist
        (unless (file-exists-p file)
          (with-temp-file file
            (insert ":PROPERTIES:\n:END:\n#+TITLE: " title "\n#+FILETAGS: :concept:seed:\n\n* Overview\n\n* Context\n* Relations\n* Applications\n* References\n")))
        ;; Transfer or create ID: use heading's existing ID if present, otherwise generate new one
        (let ((new-id (if orig-id
                          ;; Use the original heading's ID for the new file
                          (progn
                            (with-current-buffer (find-file-noselect file)
                              (goto-char (point-min))
                              (when (re-search-forward "^:PROPERTIES:" nil t)
                                (forward-line 1)
                                (insert (format ":ID: %s\n" orig-id)))
                              (save-buffer))
                            ;; Delete entire properties drawer from original heading
                            (save-excursion
                              (org-back-to-heading t)
                              (when (re-search-forward "^[ \t]*:PROPERTIES:" (save-excursion (outline-next-heading) (point)) t)
                                (let ((beg (line-beginning-position)))
                                  (when (re-search-forward "^[ \t]*:END:" (save-excursion (outline-next-heading) (point)) t)
                                    (delete-region beg (1+ (line-end-position)))))))
                            (save-buffer)
                            orig-id)
                        ;; No existing ID, generate a new one for the concept file
                        (let ((generated-id (org-id-new)))
                          (with-current-buffer (find-file-noselect file)
                            (goto-char (point-min))
                            (when (re-search-forward "^:PROPERTIES:" nil t)
                              (forward-line 1)
                              (insert (format ":ID: %s\n" generated-id)))
                            (save-buffer))
                          generated-id))))
          ;; Insert body into Overview if present
          (when (and body (not (string-empty-p (string-trim body))))
            (with-current-buffer (find-file-noselect file)
              (save-excursion
                (goto-char (point-min))
                (when (re-search-forward "^\\* Overview\\s-*$" nil t)
                  (forward-line 1)
                  ;; Clean up: remove any existing blank lines after Overview heading
                  (while (looking-at-p "^\\s-*$")
                    (delete-region (line-beginning-position) (1+ (line-end-position))))
                  ;; Insert the trimmed body content
                  (insert (string-trim body) "\n")
                  (save-buffer)))))
          ;; Insert a link to the NEW concept into the original file's *Relations section
          (save-excursion
            (goto-char (point-min))
            (if (re-search-forward "^\\* Relations\\s-*$" nil t)
                ;; Found Relations section, insert link
                (progn
                  (forward-line 1)
                  (insert (format "- [[id:%s][%s]]\n" new-id title))
                  (save-buffer))
              ;; No Relations section, create one at end of file
              (goto-char (point-max))
              (unless (bolp) (insert "\n"))
              (insert (format "* Relations\n- [[id:%s][%s]]\n" new-id title))
              (save-buffer)))
           (message "Promoted '%s' → %s (id:%s) and inserted link in original file" title file new-id)))))

** Smart Image Pasting

Smart paste function that handles both clipboard image data (screenshots) and file paths (Yazi workflow).
Automatically detects clipboard content type and routes to appropriate handler.

#+begin_src emacs-lisp
(defun my/org-smart-image-paste ()
  "Smart paste for Org using note-level UUID attach dirs.
Detects clipboard content and routes accordingly:
- file:// URI (Yazi) -> copy file into this note's UUID attach dir and insert [[file:...]]
- http(s) URL -> download via `org-download-image` into this file's attach dir
- image data in clipboard -> `org-download-clipboard`
All links inserted are file links, so inline toggling works with Org 9.8 previews."
  (interactive)
  (unless (and (derived-mode-p 'org-mode) buffer-file-name)
    (user-error "Must be in a file-backed org-mode buffer"))
  (let* (;; Try to get clipboard text with timeout (2 seconds)
         (clipboard-text (condition-case nil
                            (with-timeout (2 nil)
                              (string-trim (shell-command-to-string 
                                           "timeout 2 xclip -o -selection clipboard -t UTF8_STRING 2>/dev/null")))
                          (error nil)))
         ;; Check if clipboard contains file:// URI from Yazi
         (is-file-uri (and clipboard-text
                           (not (string-empty-p clipboard-text))
                           (string-prefix-p "file://" clipboard-text)))
         ;; Extract actual file path from URI
         (filepath (when is-file-uri
                     (url-unhex-string (substring clipboard-text 7))))
         ;; Detect web URL in clipboard
         (is-web-url (and clipboard-text
                          (not (string-empty-p clipboard-text))
                          (string-match-p "^https?://" clipboard-text)))
         ;; Validate local file from Yazi
         (is-valid-file (and filepath
                              (file-exists-p filepath)
                              (file-regular-p filepath)
                              (member (downcase (or (file-name-extension filepath) ""))
                                      '("png" "jpg" "jpeg" "gif" "webp" "bmp" "pdf" "md" "txt" "org")))))
    (cond
      ;; Yazi file:// URI workflow
      (is-valid-file
       (my/org-copy-file-into-file-attach-dir filepath))
      ;; Remote URL in clipboard (download explicitly to avoid package timing issues)
      (is-web-url
       (condition-case err
           (my/org-download-url-into-file-attach-dir clipboard-text)
         (error
            (user-error "Clipboard URL is not a downloadable image/PDF (or fetch failed): %s"
                        (error-message-string err)))))
      ;; Screenshot workflow
      (t
        (condition-case err
            (progn
              (my/org-paste-clipboard-image-to-file-attach-dir))
          (error
            (user-error "No image found in clipboard. Please copy an image (screenshot) or file from Yazi first. Error: %s" 
                       (error-message-string err))))))))

(defun my/org--unique-destination-file (dir filename)
  "Return a non-existing path in DIR for FILENAME."
  (let* ((ext (or (file-name-extension filename t) ""))
         (base (file-name-sans-extension filename))
         (candidate (expand-file-name filename dir))
         (n 1))
    (while (file-exists-p candidate)
      (setq candidate (expand-file-name (format "%s_%d%s" base n ext) dir)
            n (1+ n)))
    candidate))

(defun my/org-copy-file-into-file-attach-dir (filepath)
  "Copy FILEPATH into this note's file-based attach dir and insert a file link."
  (unless (and (derived-mode-p 'org-mode) buffer-file-name)
    (user-error "Must be in a file-backed org-mode buffer"))
  (let* ((dest-dir (my/org-file-attach-dir buffer-file-name))

                      (my/org--unique-destination-file dest-dir src-name)))

    (copy-file filepath dest-path nil)
    (unless (bolp) (insert "\n"))
    (insert (org-link-make-string (concat "file:" (org-link-escape rel-path))) "\n")

(defun my/org--buffer-prefix-p (s)
  "Return non-nil if current buffer starts with byte string S."
  (and (>= (- (point-max) (point-min)) (length s))
       (string= s (buffer-substring-no-properties (point-min)
                                                  (+ (point-min) (length s))))))

(defun my/org--detect-image-mime-from-buffer ()
  "Best-effort MIME detection for image bytes in current (unibyte) buffer."
  (cond
   ;; PNG
   ((my/org--buffer-prefix-p "\x89PNG\r\n\x1a\n") "image/png")
   ;; JPEG
   ((my/org--buffer-prefix-p "\xff\xd8") "image/jpeg")
   ;; GIF
   ((or (my/org--buffer-prefix-p "GIF87a")
        (my/org--buffer-prefix-p "GIF89a"))
    "image/gif")
   ;; BMP
   ((my/org--buffer-prefix-p "BM") "image/bmp")
   ;; WEBP: RIFF....WEBP
   ((and (my/org--buffer-prefix-p "RIFF")
         (>= (- (point-max) (point-min)) 12)
         (string= "WEBP"
                  (buffer-substring-no-properties (+ (point-min) 8)
                                                  (+ (point-min) 12))))
    "image/webp")
   ;; SVG (text XML)
   ((save-excursion
      (goto-char (point-min))
      (re-search-forward "<svg\\b" (min (point-max) (+ (point-min) 4096)) t))
    "image/svg+xml")
   (t nil)))

(defun my/org-paste-clipboard-image-to-file-attach-dir ()
  "Paste clipboard image into this note's file attach dir and insert a file link.
This avoids creating heading-level ID drawers."
  (unless (and (derived-mode-p 'org-mode) buffer-file-name)
    (user-error "Must be in a file-backed org-mode buffer"))
  (let* ((dest-dir (my/org-file-attach-dir buffer-file-name))
         (tsfmt (if (boundp 'org-download-timestamp)
                    org-download-timestamp
                  "_%Y%m%d_%H%M%S"))
         (targets (or (and (executable-find "xclip")
                           (split-string
                            (shell-command-to-string "xclip -selection clipboard -o -t TARGETS 2>/dev/null")
                            "\n" t))
                      (and (executable-find "wl-paste")
                           (split-string
                            (shell-command-to-string "wl-paste --list-types 2>/dev/null")
                            "\n" t))
                      '()))
         (target-mime (seq-find (lambda (mime) (string-prefix-p "image/" mime)) targets))
         (preferred-cmd (cond
                         ((and target-mime (executable-find "xclip"))
                          (list "xclip" "-selection" "clipboard" "-o" "-t" target-mime))
                         ((and target-mime (executable-find "wl-paste"))
                          (list "wl-paste" "-t" target-mime))
                         (t nil)))
         (fallback-cmd (cond
                        ((executable-find "xclip")
                         '("xclip" "-selection" "clipboard" "-o"))
                        ((executable-find "wl-paste")
                         '("wl-paste"))
                        (t nil)))
         (image-bytes nil)
         (mime nil))
    (unless (or (executable-find "xclip") (executable-find "wl-paste"))
      (user-error "No clipboard image tool found (need wl-paste or xclip)"))
    (with-temp-buffer
      (set-buffer-multibyte nil)
      (let ((got-bytes nil))
        (when preferred-cmd
          (setq got-bytes
                (= 0 (apply #'call-process (car preferred-cmd) nil t nil (cdr preferred-cmd))))
          (unless got-bytes
            (erase-buffer)))
        (unless got-bytes
          (erase-buffer)
          (unless (and fallback-cmd
                       (= 0 (apply #'call-process (car fallback-cmd) nil t nil (cdr fallback-cmd))))
            (user-error "No image in clipboard or clipboard command failed"))))
      (when (= (buffer-size) 0)
        (user-error "Clipboard is empty"))
      (setq mime (or target-mime (my/org--detect-image-mime-from-buffer)))
      (setq image-bytes
            (string-as-unibyte
             (buffer-substring-no-properties (point-min) (point-max)))))

    (let* ((ext (pcase mime
                  ((or "image/jpeg" "image/jpg") "jpg")
                  ("image/webp" "webp")
                  ("image/gif" "gif")
                  ("image/bmp" "bmp")
                  ("image/svg+xml" "svg")
                  ("image/png" "png")
                  (_ nil))))
      (unless ext
        (user-error "No image detected in clipboard"))
      (let* ((filename (format "%sscreenshot.%s" (format-time-string tsfmt) ext))
             (dest-path (progn
                          (make-directory dest-dir t)
                          (my/org--unique-destination-file dest-dir filename)))
             (rel-path (file-relative-name dest-path (file-name-directory buffer-file-name))))
        (with-temp-file dest-path
          (set-buffer-multibyte nil)
          (insert image-bytes))
        (ignore-errors (image-flush dest-path))
        (unless (bolp) (insert "\n"))
        (insert (org-link-make-string (concat "file:" (org-link-escape rel-path))) "\n")
        (my/org-refresh-inline-previews (point-min) (point-max))
        (message "Pasted clipboard image to %s" rel-path)))))

(defun my/org-remove-empty-id-drawers-in-current-buffer ()
  "Remove empty property drawers under headings in current Org buffer.
Keeps top-level node metadata drawer intact for org-roam."
  (interactive)
  (unless (derived-mode-p 'org-mode)
    (user-error "Not in org-mode"))
  (let ((removed 0))
    (save-excursion
      (goto-char (point-min))
      (while (re-search-forward "^\\*+ " nil t)
        (save-excursion
          (forward-line 1)
          (when (looking-at-p "^[ \t]*:PROPERTIES:")
            (let ((drawer-beg (point))
                  (drawer-end (save-excursion
                                (when (re-search-forward "^[ \t]*:END:[ \t]*$" nil t)
                                  (line-end-position)))))
              (when drawer-end
                (let ((content (buffer-substring-no-properties drawer-beg drawer-end)))
                  (when (not (string-match-p "^[ \t]*:PROPERTIES:[ \t\n]*:END:[ \t]*$" content))
                    (setq drawer-beg nil)))
                (when drawer-beg
                  (delete-region drawer-beg drawer-end)
                  (delete-blank-lines)
                  (setq removed (1+ removed)))))))))
    (message "Removed %d empty heading property drawers" removed)
    removed))

(defun my/org-migrate-attachment-links-in-current-buffer (&optional dry-run)
  "Migrate [[attachment:...]] links in current buffer to [[file:...]] links.
Copies files from heading-based org-attach dirs into this file's attach dir.
With DRY-RUN non-nil, only report what would change."
  (interactive "P")
  (unless (and (derived-mode-p 'org-mode) buffer-file-name)
    (user-error "Must be in a file-backed org-mode buffer"))
  (require 'org-attach)
  (let ((changes '())
        (converted 0)
        (copied 0)
        (skipped 0))
    (save-excursion
      (goto-char (point-min))
      (while (re-search-forward org-link-bracket-re nil t)
        (let ((ctx (save-match-data (org-element-context))))
          (when (and (eq (org-element-type ctx) 'link)
                     (string= (org-element-property :type ctx) "attachment"))
            (let* ((beg (org-element-property :begin ctx))
                   (end (org-element-property :end ctx))
                   (path (org-element-property :path ctx))
                   (desc (when-let ((cb (org-element-property :contents-begin ctx))
                                    (ce (org-element-property :contents-end ctx)))
                           (buffer-substring-no-properties cb ce)))
                   (src (save-excursion
                          (goto-char beg)
                          (condition-case nil
                              (org-attach-expand path)
                            (error nil)))))
              (if (and src (file-exists-p src))
                  (let* ((dest-dir (my/org-file-attach-dir buffer-file-name))
                         (dest-path (progn
                                      (make-directory dest-dir t)
                                      (my/org--unique-destination-file
                                       dest-dir (file-name-nondirectory src))))
                         (rel-path (file-relative-name dest-path (file-name-directory buffer-file-name)))
                         (new-link (org-link-make-string
                                    (concat "file:" (org-link-escape rel-path))
                                    desc)))
                    (push (list beg end src dest-path new-link) changes))
                (setq skipped (1+ skipped))))))))
    (dolist (change (sort changes (lambda (a b) (> (car a) (car b)))))
      (pcase-let ((`(,beg ,end ,src ,dest ,new-link) change))
        (setq converted (1+ converted))
        (unless dry-run
          (copy-file src dest nil)
          (setq copied (1+ copied))
          (goto-char beg)
          (delete-region beg end)
          (insert new-link))))
    (unless (or dry-run (= converted 0))
      (save-buffer))
    (message "%s: converted=%d copied=%d skipped=%d"
             (if dry-run "DRY-RUN" "Migrated") converted copied skipped)
    (list :converted converted :copied copied :skipped skipped :dry-run dry-run)))

(defun my/org-migrate-attachment-links-in-brain2 (&optional dry-run)
  "Migrate attachment links across `org-directory` recursively.
With DRY-RUN non-nil, only report totals."
  (interactive "P")
  (let* ((root (expand-file-name org-directory))
         (files (directory-files-recursively root "\\.org\\'"))
         (total-converted 0)
         (total-copied 0)
         (total-skipped 0)
         (touched 0))
    (dolist (file files)
      (with-current-buffer (find-file-noselect file)
        (let ((res (my/org-migrate-attachment-links-in-current-buffer dry-run)))
          (cl-incf total-converted (or (plist-get res :converted) 0))
          (cl-incf total-copied (or (plist-get res :copied) 0))
          (cl-incf total-skipped (or (plist-get res :skipped) 0))
          (when (> (or (plist-get res :converted) 0) 0)
            (cl-incf touched)))))
    (message "%s %d files (touched=%d): converted=%d copied=%d skipped=%d"
             (if dry-run "DRY-RUN across" "Migrated across")
             (length files) touched total-converted total-copied total-skipped)
    (list :files (length files)
          :touched touched
          :converted total-converted
          :copied total-copied
          :skipped total-skipped
          :dry-run dry-run)))

** Org-Roam Link Hover (K)

Pressing =K= on an =[[id:...]]=  link shows a floating preview of the linked
node's section content (via eldoc-box). Pressing =K= again follows the link.
Falls through to normal =K= behaviour when not on an org-roam link.

#+begin_src emacs-lisp
(defvar +org-roam--box-was-open nil
  "Non-nil when the org-roam hover frame was shown on the last K press.")
(defun +org-roam-link-at-point ()
  "Return the id: link element at point, or nil if not on one."
  (when (derived-mode-p 'org-mode)
    (let ((ctx (org-element-context)))
      (when (and (eq (org-element-type ctx) 'link)
                 (equal (org-element-property :type ctx) "id"))
        ctx))))
(defun +org-roam--plain-text (str)
  "Strip org syntax from STR, returning lightly-formatted text for hover display.
Headings are bolded. Code spans use a monospace face. Quotes get a │ prefix.
/Italics/ are italicised. Everything else is plain."
  (let ((lines (split-string str "\n"))
        (in-quote nil)
        result)
    (dolist (line lines)
      (cond
       ;; #+begin_quote → enter quote mode (drop the marker line)
       ((string-match-p "^#\\+begin_quote" line)
        (setq in-quote t))
       ;; #+end_quote → exit quote mode (drop the marker line)
       ((string-match-p "^#\\+end_quote" line)
        (setq in-quote nil))
       ;; Other #+KEYWORD lines — drop entirely
       ((string-match-p "^#\\+" line))
       ;; :PROPERTIES: / :KEY: val / :END: — drop
       ((string-match-p "^\\s-*:[A-Z_]+:" line))
        ;; Headings: strip links, then bold
        ((string-match "^\\(\\*+\\) +\\(.*\\)" line)
         (let* ((text (match-string 2 line))
                ;; [[id:...][desc]] → desc
                (text (replace-regexp-in-string
                       "\\[\\[id:[^]]+\\]\\[\\([^]]+\\)\\]\\]" "\\1" text))
                ;; bare [[id:...]] → remove
                (text (replace-regexp-in-string "\\[\\[id:[^]]+\\]\\]" "" text))
                ;; [[anything][desc]] → desc
                (text (replace-regexp-in-string
                       "\\[\\[[^]]+\\]\\[\\([^]]+\\)\\]\\]" "\\1" text)))
           (push (propertize (string-trim text) 'face 'bold) result)
           (push "" result)))
       ;; All other lines (normal text or inside a quote block)
       (t
        (let ((s line))
          ;; [[id:...][desc]] → desc
          (setq s (replace-regexp-in-string
                   "\\[\\[id:[^]]+\\]\\[\\([^]]+\\)\\]\\]" "\\1" s))
          ;; [[id:...]] bare → remove
          (setq s (replace-regexp-in-string "\\[\\[id:[^]]+\\]\\]" "" s))
          ;; [[anything][desc]] → desc
          (setq s (replace-regexp-in-string
                   "\\[\\[[^]]+\\]\\[\\([^]]+\\)\\]\\]" "\\1" s))
          ;; =code= and ~verbatim~ → monospace face
          (setq s (replace-regexp-in-string
                   "[=~]\\([^=~\n]+\\)[=~]"
                   (lambda (m)
                     (propertize (match-string 1 m) 'face 'fixed-pitch))
                   s))
          ;; *bold* → bold
          (setq s (replace-regexp-in-string
                   "\\*\\([^*\n]+\\)\\*"
                   (lambda (m)
                     (propertize (match-string 1 m) 'face 'bold))
                   s))
          ;; /italic/ → italic face
          (setq s (replace-regexp-in-string
                   "/\\([^/\n]+\\)/"
                   (lambda (m)
                     (propertize (match-string 1 m) 'face 'italic))
                   s))
          ;; _underline_ → strip markers, keep text
          (setq s (replace-regexp-in-string "_\\([^_\n]+\\)_" "\\1" s))
          ;; Quote block lines get a │ prefix
          (when in-quote
            (setq s (concat (propertize "│ " 'face 'shadow) s)))
          (push s result)))))
    ;; Reverse (we pushed), join, collapse excess blank lines
    (let ((out (string-join (nreverse result) "\n")))
      (setq out (replace-regexp-in-string "\n\n\n+" "\n\n" out))
      (string-trim out))))
(defun +org-roam-file-preview (file &optional n-lines)
  "Return the first N-LINES (default 20) of plain-text content from FILE.
Skips #+KEYWORD preamble and property drawers, strips org markup."
  (let ((n (or n-lines 20)))
    (org-roam-with-temp-buffer file
      (org-with-wide-buffer
       (goto-char (point-min))
       ;; Skip #+KEYWORD lines and blank lines at top
       (while (and (not (eobp))
                   (looking-at-p "^\\(#\\+\\|$\\|:PROPERTIES:\\)"))
         (if (looking-at-p "^:PROPERTIES:")
             (re-search-forward "^:END:\n?" nil t)
           (forward-line 1)))
       (let* ((beg (point))
              (end (save-excursion (forward-line n) (point)))
              (raw (buffer-substring-no-properties beg end)))
         (+org-roam--plain-text raw))))))
(defun +org-roam-K-at-point (_identifier)
  "Hover-preview the org-roam node linked at point via eldoc-box.
First K: show a floating preview of the linked heading's section.
          For file-level nodes, shows the first lines of content.
Second K: follow the link (like RET).
Returns nil when point is not on an id: link, allowing fallthrough."
  (when-let ((link (+org-roam-link-at-point)))
    (if +org-roam--box-was-open
        (progn
          (setq +org-roam--box-was-open nil)
          (eldoc-box-quit-frame)
          (org-open-at-point))
      (let* ((id   (org-element-property :path link))
             (node (org-roam-node-from-id id))
             (loc  (org-id-find id)))
        (if (not loc)
            (user-error "org-roam: can't find node %s" id)
          (let* ((file  (car loc))
                 (pt    (cdr loc))
                 (title (if node (org-roam-node-title node) id))
                 (level (if node (org-roam-node-level node) 1))
                 ;; File-level nodes (level 0): use first-lines preview.
                 ;; Heading nodes (level > 0): use the section content.
                 ;; Both are stripped to plain text.
                 (body  (if (= level 0)
                            (+org-roam-file-preview file)
                          (+org-roam--plain-text
                           (or (org-roam-preview-get-contents file pt) ""))))
                 (str   (concat (propertize title 'face 'bold) "\n\n"
                                (if (or (null body) (string-empty-p body))
                                    "(no preview available)"
                                  body))))
            (setq +org-roam--box-was-open t)
            (advice-add 'eldoc-box-quit-frame :after
                        (lambda (&rest _) (setq +org-roam--box-was-open nil))
                        '((name . +org-roam--reset-box-flag)))
            (let ((eldoc-box-position-function
                   eldoc-box-at-point-position-function))
              (eldoc-box--display str))
            ;; Mirror what eldoc-box-help-at-point does: record point and
            ;; start the cleanup timer so the frame dismisses on movement.
            (setq eldoc-box--help-at-point-last-point (point))
            (run-with-timer 0.1 nil #'eldoc-box--help-at-point-cleanup)
            (when eldoc-box-clear-with-C-g
              (advice-add #'keyboard-quit :before #'eldoc-box-quit-frame))))))
    t))
(after! (org-roam eldoc-box)
  (set-lookup-handlers! 'org-mode
    :documentation #'+org-roam-K-at-point))

;; More robust force-expand for yasnippet while completion UIs are active.
(defun my/force-yas-expand ()
  "Abort completion popups and force a yasnippet expansion at/near point.
Tries, in order:
  1. yas-expand at point
  2. yas-expand at the end of the nearest preceding symbol
  3. scan backwards for a preceding symbol and try expanding there
This tries to survive Corfu/Company/Copilot overlays that steal keys or move point."
  (interactive)
  (let ((expanded nil)
        (orig-point (point)))
    ;; Abort/close common completion UIs (best-effort)
    (when (and (boundp 'company-mode) (fboundp 'company--active-p) (company--active-p))
      (ignore-errors (company-abort)))
    (when (and (boundp 'corfu-mode) (fboundp 'corfu-quit))
      (ignore-errors (corfu-quit)))
    ;; Copilot/ghost overlays: try known cleanup functions (best-effort)
    (when (bound-and-true-p copilot-mode)
      (cond
       ((fboundp 'copilot-clear-overlay) (ignore-errors (copilot-clear-overlay)))
       ((fboundp 'copilot--dismiss)     (ignore-errors (copilot--dismiss)))
       ((fboundp 'copilot-complete)     (ignore-errors (copilot-complete)))))
    ;; 1) Try expanding at point
    (when (fboundp 'yas-expand)
      (setq expanded (condition-case _ (yas-expand) (error nil))))
    ;; 2) If not expanded, try the end of the current/previous symbol
    (unless expanded
      (save-excursion
        (let ((end (point)))
          (skip-chars-backward "A-Za-z0-9_-")
          (when (< (point) end)
            (goto-char end)
            (setq expanded (condition-case _ (yas-expand) (error nil)))))))
    ;; 3) If still not expanded, scan backwards for a candidate symbol and try expanding there
    (unless expanded
      (save-excursion
        (when (search-backward-regexp "\\_<[A-Za-z0-9_+-]+\\_>" nil t)
          (goto-char (match-end 0))
          (setq expanded (condition-case _ (yas-expand) (error nil))))))
    (unless expanded
      (goto-char orig-point)           ; restore point if we moved
      (message "my/force-yas-expand: no snippet expansion found"))
    expanded))

;; Bind the function into known completion/overlay maps so it wins when popups are active
(dolist (m '(company-active-map
             company-mode-map
             corfu-map
             corfu-popupinfo-map
             copilot-completion-map
             aya-mode-map))
